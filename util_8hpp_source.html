87 inline DWORD enableVtMode()
88 {
89  // Set output mode to handle virtual terminal sequences
90  const HANDLE hOut = GetStdHandle(STD_OUTPUT_HANDLE);
91  if (hOut == INVALID_HANDLE_VALUE)
92  return -1;
93 
94  DWORD dwMode = 0;
95  if (!GetConsoleMode(hOut, &dwMode))
96  return -1;
97  const DWORD dwModeOrig = dwMode;
98 
99  dwMode |= ENABLE_VIRTUAL_TERMINAL_PROCESSING;
100  if (!SetConsoleMode(hOut, dwMode))

102  return dwModeOrig;
103 }

194 constexpr bool isNumber(const std::string_view& s)
195 {
196  if (s.empty())
197  return false;
198  int decimalPointCount = 0, minusCount = 0;
199 
200  for (const char c : s)
201  {
202  if (c == '.')
203  {
204  if (decimalPointCount > 0)
205  return false;
206  decimalPointCount++;
207  }
208  else if (c == '-')
209  {
210  if (minusCount > 0)
211  return false;
212  minusCount++;
213  }
214  else if (not isdigit(c))
215  return false;
216  }
217  return true;
218 }

247 template<typename CharT>
248 auto split(std::basic_string_view<CharT> s, const CharT separator)
249 {
250  std::vector<decltype(s)> result;
251  auto left = s.begin();
252  for (auto it = left; it != s.end(); ++it)
253  {
254  if (*it == separator)
255  {
256  result.emplace_back(&*left, it - left);
257  left = it + 1;
258  }
259  }
260  if (left != s.end())
261  result.emplace_back(&*left, s.end() - left);
262  return result;
263 }

297 {

299  typename std::basic_string<CharT>::size_type count = 0;

305  if (replacement != '\0')
