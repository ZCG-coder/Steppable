92 inline DWORD enableVtMode()
93 {
94  // Set output mode to handle virtual terminal sequences
95  const HANDLE hOut = GetStdHandle(STD_OUTPUT_HANDLE);
96  if (hOut == INVALID_HANDLE_VALUE)
97  return -1;
98 
99  DWORD dwMode = 0;
100  if (!GetConsoleMode(hOut, &dwMode))
101  return -1;
102  const DWORD dwModeOrig = dwMode;
103 
104  dwMode |= ENABLE_VIRTUAL_TERMINAL_PROCESSING;
105  if (!SetConsoleMode(hOut, dwMode))
106  return -1;
107  return dwModeOrig;
108 }

198 constexpr bool isNumber(const std::string_view& s)
199 {
200  if (s.empty())
201  return false;
202  int decimalPointCount = 0, minusCount = 0;
203 
204  for (const char c : s)
205  {
206  if (c == '.')
207  {
208  if (decimalPointCount > 0)
209  return false;
210  decimalPointCount++;
211  }
212  else if (c == '-')
213  {
214  if (minusCount > 0)
215  return false;
216  minusCount++;
217  }
218  else if (not isdigit(c))
219  return false;
220  }
221  return true;
222 }

251 template<typename CharT>
252 auto split(std::basic_string_view<CharT> s, const CharT separator)
253 {
254  std::vector<decltype(s)> result;
255  auto left = s.begin();
256  for (auto it = left; it != s.end(); ++it)
257  {
258  if (*it == separator)
259  {
260  result.emplace_back(&*left, it - left);
261  left = it + 1;
262  }
263  }
264  if (left != s.end())
265  result.emplace_back(&*left, s.end() - left);
266  return result;
267 }

301 {
302  std::basic_string<CharT> out = s;
303  typename std::basic_string<CharT>::size_type count = 0;

305  {
306  out.pop_back();

308  }
309  if (replacement != '\0')
310  out += std::basic_string<CharT>(count, replacement);
311  return out;
