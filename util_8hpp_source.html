81  inline DWORD enableVtMode()
82  {
83  // Set output mode to handle virtual terminal sequences
84  const HANDLE hOut = GetStdHandle(STD_OUTPUT_HANDLE);
85  if (hOut == INVALID_HANDLE_VALUE)
86  return -1;
87 
88  DWORD dwMode = 0;
89  if (!GetConsoleMode(hOut, &dwMode))
90  return -1;
91  const DWORD dwModeOrig = dwMode;
92 
93  dwMode |= ENABLE_VIRTUAL_TERMINAL_PROCESSING;
94  if (!SetConsoleMode(hOut, dwMode))
95  return -1;
96  return dwModeOrig;
97  }

238  constexpr bool isNumber(const std::string_view& s)
239  {
240  if (s.empty())
241  return false;
242  int decimalPointCount = 0, minusCount = 0;
243 
244  for (const char c : s)
245  {
246  if (c == '.')
247  {
248  if (decimalPointCount > 0)
249  return false;
250  decimalPointCount++;
251  }
252  else if (c == '-')
253  {
254  if (minusCount > 0)
255  return false;
256  minusCount++;
257  }
258  else if (not isdigit(c))
259  return false;
260  }
261  return true;
262  }

343  {
344  std::vector<decltype(s)> substrings;
345  std::basic_stringstream<CharT> ss(s);
346  decltype(s) token;

351  return substrings;

361  template<typename CharT>
362  auto split(std::basic_string_view<CharT> s, const CharT separator)
363  {
364  std::vector<decltype(s)> result;
365  auto left = s.begin();
366  for (auto it = left; it != s.end(); ++it)
367  {
368  if (*it == separator)
369  {
370  result.emplace_back(&*left, it - left);
371  left = it + 1;
372  }
373  }
374  if (left != s.end())
375  result.emplace_back(&*left, s.end() - left);
376  return result;
377  }

393  template<typename CharT>
394  auto rReplace(const std::basic_string<CharT> s, const CharT t, const CharT replacement = '\0')
395  {
396  std::basic_string<CharT> out = s;
397  typename std::basic_string<CharT>::size_type count = 0;
398  while (out.back() == t)
399  {
400  out.pop_back();
401  ++count;
402  }
403  if (replacement != '\0')
404  out += std::basic_string<CharT>(count, replacement);
405  return out;
406  }

419  template<typename CharT>
420  auto lReplace(const std::basic_string<CharT> s, const CharT t, const CharT replacement = '\0')
421  {
422  std::basic_string<CharT> out = s;
423  typename std::basic_string<CharT>::size_type count = 0;
424  while (out[0] == t)
425  {
426  out.erase(out.begin());
427  ++count;
428  }
429  if (replacement != '\0')
430  out = std::basic_string<CharT>(count, replacement) + out;
431  return out;
