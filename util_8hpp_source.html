112 inline DWORD enableVtMode()
113 {
114 // Set output mode to handle virtual terminal sequences
115 const HANDLE hOut = GetStdHandle(STD_OUTPUT_HANDLE);
116 if (hOut == INVALID_HANDLE_VALUE)
117 return -1;
118
119 DWORD dwMode = 0;
120 if (!GetConsoleMode(hOut, &dwMode))
121 return -1;
122 const DWORD dwModeOrig = dwMode;
123
124 dwMode |= ENABLE_VIRTUAL_TERMINAL_PROCESSING;
125 if (!SetConsoleMode(hOut, dwMode))
126 return -1;
127 return dwModeOrig;
128 }
129
136 inline bool restoreVtMode(const DWORD dwModeOrig)
137 {
138 const HANDLE hOut = GetStdHandle(STD_OUTPUT_HANDLE);
139 if (hOut == INVALID_HANDLE_VALUE)
140 return false;
141
142 if (!SetConsoleMode(hOut, dwModeOrig))
143 return false;
144 return true;
145 }

433 template<typename CharT>
434 auto split(std::basic_string<CharT> s, const CharT separator)
435 {
436 std::vector<decltype(s)> substrings;
437 std::basic_stringstream<CharT> ss(s);
438 decltype(s) token;
439
440 while (getline(ss, token, separator))
441 substrings.push_back(token);
442
443 return substrings;
444 }

453 template<typename CharT>
454 auto split(std::basic_string_view<CharT> s, const CharT separator)
455 {
456 std::vector<decltype(s)> result;
457 auto left = s.begin();
458 for (auto it = left; it != s.end(); ++it)
459 {
460 if (*it == separator)
461 {
462 result.emplace_back(&*left, it - left);
463 left = it + 1;
464 }
465 }
466 if (left != s.end())
467 result.emplace_back(&*left, s.end() - left);
468 return result;
469 }

485 template<typename CharT>
486 auto rReplace(const std::basic_string<CharT> s, const CharT t, const CharT replacement = '\0')
487 {
488 std::basic_string<CharT> out = s;
489 typename std::basic_string<CharT>::size_type count = 0;
490 while (out.back() == t)
491 {
492 out.pop_back();
493 ++count;
494 }
495 if (replacement != '\0')
496 out += std::basic_string<CharT>(count, replacement);
497 return out;
498 }

511 template<typename CharT>
512 auto lReplace(const std::basic_string<CharT> s, const CharT t, const CharT replacement = '\0')
513 {
514 std::basic_string<CharT> out = s;
515 typename std::basic_string<CharT>::size_type count = 0;
516 while (out[0] == t)
517 {
518 out.erase(out.begin());
519 ++count;
520 }
521 if (replacement != '\0')
522 out = std::basic_string<CharT>(count, replacement) + out;
523 return out;
524 }
