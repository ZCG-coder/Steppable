100 inline DWORD enableVtMode()
101 {
102 // Set output mode to handle virtual terminal sequences
103 const HANDLE hOut = GetStdHandle(STD_OUTPUT_HANDLE);
104 if (hOut == INVALID_HANDLE_VALUE)
105 return -1;
106
107 DWORD dwMode = 0;
108 if (!GetConsoleMode(hOut, &dwMode))
109 return -1;
110 const DWORD dwModeOrig = dwMode;
111
112 dwMode |= ENABLE_VIRTUAL_TERMINAL_PROCESSING;
113 if (!SetConsoleMode(hOut, dwMode))
114 return -1;
115 return dwModeOrig;
116 }
117
124 inline bool restoreVtMode(const DWORD dwModeOrig)
125 {
126 const HANDLE hOut = GetStdHandle(STD_OUTPUT_HANDLE);
127 if (hOut == INVALID_HANDLE_VALUE)
128 return false;
129
130 if (!SetConsoleMode(hOut, dwModeOrig))
131 return false;
132 return true;
133 }

354 template<typename CharT>
355 auto split(std::basic_string<CharT> s, const CharT separator)
356 {
357 std::vector<decltype(s)> substrings;
358 std::basic_stringstream<CharT> ss(s);
359 decltype(s) token;
360
361 while (getline(ss, token, separator))
362 substrings.push_back(token);
363
364 return substrings;
365 }

374 template<typename CharT>
375 auto split(std::basic_string_view<CharT> s, const CharT separator)
376 {
377 std::vector<decltype(s)> result;
378 auto left = s.begin();
379 for (auto it = left; it != s.end(); ++it)
380 {
381 if (*it == separator)
382 {
383 result.emplace_back(&*left, it - left);
384 left = it + 1;
385 }
386 }
387 if (left != s.end())
388 result.emplace_back(&*left, s.end() - left);
389 return result;
390 }

406 template<typename CharT>
407 auto rReplace(const std::basic_string<CharT> s, const CharT t, const CharT replacement = '\0')
408 {
409 std::basic_string<CharT> out = s;
410 typename std::basic_string<CharT>::size_type count = 0;
411 while (out.back() == t)
412 {
413 out.pop_back();
414 ++count;
415 }
416 if (replacement != '\0')
417 out += std::basic_string<CharT>(count, replacement);
418 return out;
419 }

432 template<typename CharT>
433 auto lReplace(const std::basic_string<CharT> s, const CharT t, const CharT replacement = '\0')
434 {
435 std::basic_string<CharT> out = s;
436 typename std::basic_string<CharT>::size_type count = 0;
437 while (out[0] == t)
438 {
439 out.erase(out.begin());
440 ++count;
441 }
442 if (replacement != '\0')
443 out = std::basic_string<CharT>(count, replacement) + out;
444 return out;
445 }
