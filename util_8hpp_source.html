99 inline DWORD enableVtMode()
100 {
101 // Set output mode to handle virtual terminal sequences
102 const HANDLE hOut = GetStdHandle(STD_OUTPUT_HANDLE);
103 if (hOut == INVALID_HANDLE_VALUE)
104 return -1;
105
106 DWORD dwMode = 0;
107 if (!GetConsoleMode(hOut, &dwMode))
108 return -1;
109 const DWORD dwModeOrig = dwMode;
110
111 dwMode |= ENABLE_VIRTUAL_TERMINAL_PROCESSING;
112 if (!SetConsoleMode(hOut, dwMode))
113 return -1;
114 return dwModeOrig;
115 }
116
123 inline bool restoreVtMode(const DWORD dwModeOrig)
124 {
125 const HANDLE hOut = GetStdHandle(STD_OUTPUT_HANDLE);
126 if (hOut == INVALID_HANDLE_VALUE)
127 return false;
128
129 if (!SetConsoleMode(hOut, dwModeOrig))
130 return false;
131 return true;
132 }

360 template<typename CharT>
361 auto split(std::basic_string<CharT> s, const CharT separator)
362 {
363 std::vector<decltype(s)> substrings;
364 std::basic_stringstream<CharT> ss(s);
365 decltype(s) token;
366
367 while (getline(ss, token, separator))
368 substrings.push_back(token);
369
370 return substrings;
371 }

380 template<typename CharT>
381 auto split(std::basic_string_view<CharT> s, const CharT separator)
382 {
383 std::vector<decltype(s)> result;
384 auto left = s.begin();
385 for (auto it = left; it != s.end(); ++it)
386 {
387 if (*it == separator)
388 {
389 result.emplace_back(&*left, it - left);
390 left = it + 1;
391 }
392 }
393 if (left != s.end())
394 result.emplace_back(&*left, s.end() - left);
395 return result;
396 }

412 template<typename CharT>
413 auto rReplace(const std::basic_string<CharT> s, const CharT t, const CharT replacement = '\0')
414 {
415 std::basic_string<CharT> out = s;
416 typename std::basic_string<CharT>::size_type count = 0;
417 while (out.back() == t)
418 {
419 out.pop_back();
420 ++count;
421 }
422 if (replacement != '\0')
423 out += std::basic_string<CharT>(count, replacement);
424 return out;
425 }

438 template<typename CharT>
439 auto lReplace(const std::basic_string<CharT> s, const CharT t, const CharT replacement = '\0')
440 {
441 std::basic_string<CharT> out = s;
442 typename std::basic_string<CharT>::size_type count = 0;
443 while (out[0] == t)
444 {
445 out.erase(out.begin());
446 ++count;
447 }
448 if (replacement != '\0')
449 out = std::basic_string<CharT>(count, replacement) + out;
450 return out;
451 }
