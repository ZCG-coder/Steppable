92 inline DWORD enableVtMode()
93 {
94  // Set output mode to handle virtual terminal sequences
95  const HANDLE hOut = GetStdHandle(STD_OUTPUT_HANDLE);
96  if (hOut == INVALID_HANDLE_VALUE)
97  return -1;
98 
99  DWORD dwMode = 0;
100  if (!GetConsoleMode(hOut, &dwMode))
101  return -1;
102  const DWORD dwModeOrig = dwMode;
103 
104  dwMode |= ENABLE_VIRTUAL_TERMINAL_PROCESSING;
105  if (!SetConsoleMode(hOut, dwMode))
106  return -1;
107  return dwModeOrig;
108 }

232 constexpr bool isNumber(const std::string_view& s)
233 {
234  if (s.empty())
235  return false;
236  int decimalPointCount = 0, minusCount = 0;
237 
238  for (const char c : s)
239  {
240  if (c == '.')
241  {
242  if (decimalPointCount > 0)
243  return false;
244  decimalPointCount++;
245  }
246  else if (c == '-')
247  {
248  if (minusCount > 0)
249  return false;
250  minusCount++;
251  }
252  else if (not isdigit(c))
253  return false;
254  }
255  return true;
256 }

285 template<typename CharT>
286 auto split(std::basic_string_view<CharT> s, const CharT separator)
287 {
288  std::vector<decltype(s)> result;
289  auto left = s.begin();
290  for (auto it = left; it != s.end(); ++it)
291  {
292  if (*it == separator)
293  {
294  result.emplace_back(&*left, it - left);
295  left = it + 1;
296  }
297  }
298  if (left != s.end())
299  result.emplace_back(&*left, s.end() - left);
300  return result;
301 }

336 template<typename CharT>
337 auto rReplace(const std::basic_string<CharT> s, const CharT t, const CharT replacement = '\0')
338 {
339  std::basic_string<CharT> out = s;
340  typename std::basic_string<CharT>::size_type count = 0;
341  while (out.back() == t)
342  {
343  out.pop_back();
344  ++count;
345  }
346  if (replacement != '\0')
347  out += std::basic_string<CharT>(count, replacement);
348  return out;
349 }

364 {
365  std::basic_string<CharT> out = s;
366  typename std::basic_string<CharT>::size_type count = 0;

369  out.erase(out.begin());

372  if (replacement != '\0')
373  out = std::basic_string<CharT>(count, replacement) + out;
