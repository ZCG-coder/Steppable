538 template<typename Pred>
539 void loop(const std::string& times, Pred predicate)
540 {
541 // We're done already!
542 if (times == "0")
543 return;
544 if (times == "1")
545 return predicate("1");
546
547 std::string current = "0";
548 auto result = compare(current, times, 0);
549
550 while (result == "0")
551 {
552 try
553 {
554 predicate(current);
555 }
556 catch (std::exception& e)
557 {
558 output::error("loop", "Exception occurred in predicate."s);
559 output::error("loop", "Exception message: {0}"s, { e.what() });
560 }
561 current = add(current, "1", 0);
562 result = compare(current, times, 0);
563 }
564 }
