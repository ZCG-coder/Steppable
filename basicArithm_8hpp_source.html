189  template<typename Pred>
190  void loop(const std::string_view& times, Pred predicate)
191  {
192  // We're done already!
193  if (times == "0")
194  return;
195  if (times == "1")
196  return predicate("1");
197 
198  std::string current = "0";
199  auto result = compare(current, times, 0);
200 
201  while (result == "0")
202  {
203  try
204  {
205  predicate(current);
206  }
207  catch (std::exception& e)
208  {
209  output::error("loop", "Exception occurred in predicate."s);
210  output::error("loop", "Exception message: %s"s, e.what());
211  }
212  current = add(current, "1", 0);
213  result = compare(current, times, 0);
214  }
215  }
