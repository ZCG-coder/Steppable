63 template<__internals::utils::StringLiteral fnName>
64 [[nodiscard]] size_t determinePrec(const Number& rhs) const
65 {
66 size_t usePrec = 0;
67 if (mode == RoundingMode::USE_MAXIMUM_PREC)
68 usePrec = std::max(prec, rhs.prec);
69 else if (mode == RoundingMode::USE_MINIMUM_PREC)
70 usePrec = std::min(prec, rhs.prec);
71 else if (mode == RoundingMode::USE_CURRENT_PREC)
72 usePrec = prec;
73 else if (mode == RoundingMode::USE_OTHER_PREC)
74 usePrec = rhs.prec;
75 else if (mode == RoundingMode::DISCARD_ALL_DECIMALS)
76 usePrec = 0;
77 else
78 {
79 usePrec = 0;
80 output::warning(std::string(fnName.value), "Invalid precision specified"s);
81 }
82
83 return usePrec;
84 }
